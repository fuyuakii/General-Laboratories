\textbf{Dedução da Equação 3:}
\begin{align*}
\log (x) &= \log (\frac{a }{2} . t^2) \quad (1) \\
\log (x) &= \log \frac{a}{2} + \log (t^2) \quad (2) \\
\log (x) &= \log (\frac{a}{2}) + 2 \log (t) \\
-2\log(t) + \log (x) &= \log (\frac{a}{2}) + 2 \log (t) - 2 \log (t) \\
\log (x) - {2 \log (t)} &= \log_{10} (\frac{a}{2}) \quad (3) \\
\frac{a}{2} &= 10^{-0,0158} \\
a &= 2 \cdot 10^{-0,0158} \\
a &= 1,92855 \ \  \text{m/s}^2
\end{align*}

    \item (3) \(\log_b (a) = c \iff a = (b^c)\)
